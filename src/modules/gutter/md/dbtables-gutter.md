  project_name text null,
  customer text null,
  project_address text null,
  status_id bigint null,
  date date null,
  trip_id bigint null,
  manufacturer_id bigint null,
  discount_id bigint null,
  request_link text null,
  created_at timestamp with time zone null,
  updated_at timestamp with time zone null,

  cstm_trip_rate numeric null,
  cstm_manufacturer_rate numeric null,
  cstm_discount_percentage numeric null,
  cstm_leaf_guard_price numeric null,
  deposit_percent numeric null,

  constraint fk_projects_discount foreign KEY (discount_id) references gtr_s_discounts (discount_id) on delete set null,
  constraint fk_projects_leaf_guard foreign KEY (leaf_guard_id) references gtr_s_leaf_guards (leaf_guard_id) on delete set null,

  constraint fk_projects_manufacturer foreign KEY (manufacturer_id) references gtr_s_manufacturers (manufacturer_id) on delete RESTRICT
) TABLESPACE pg_default;
